    **onDragStart handler:**
    ```typescript
    function handleDragStart(event: DragStartEvent) {
      const data = event.active.data.current as { card: Card; fromZone: string; fromId: string };
      dragDataRef.current = data; // Capture before optimistic removal (Pitfall 3)
      setActiveCard(data.card);
      setDragging(true);
    }
    ```

    **onDragEnd handler:**
    ```typescript
    function handleDragEnd(event: DragEndEvent) {
      setActiveCard(null);
      setDragging(false);

      if (!event.over || !dragDataRef.current) {
        dragDataRef.current = null;
        return; // No valid drop target — card snaps back via buffer flush
      }

      const { card, fromZone, fromId } = dragDataRef.current;
      const overData = event.over.data.current as { toZone: string; toId: string };

      // Block drops on opponent hands (D-14)
      if (!overData || !overData.toZone) {
        dragDataRef.current = null;
        return;
      }

      sendAction({
        type: 'MOVE_CARD',
        cardId: card.id,
        fromZone: fromZone as 'hand' | 'pile',
        fromId,
        toZone: overData.toZone as 'hand' | 'pile',
        toId: overData.toId,
      });

      dragDataRef.current = null;
    }
    ```

    **onDragCancel handler:**
    ```typescript
    function handleDragCancel() {
      setActiveCard(null);
      setDragging(false);
      dragDataRef.current = null;
    }
    ```

    <DndContext
      collisionDetection={closestCenter}
      onDragStart={handleDragStart}
      onDragEnd={handleDragEnd}
      onDragCancel={handleDragCancel}
    >
      <BoardView gameState={gameState} playerId={playerId} sendAction={sendAction} />
      {createPortal(
