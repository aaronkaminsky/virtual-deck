**Create src/hooks/usePlayerId.ts:**

```typescript
import { nanoid } from 'nanoid';

const STORAGE_KEY = 'playerId';

export function getOrCreatePlayerId(): string {
  const existing = localStorage.getItem(STORAGE_KEY);
  if (existing) return existing;
  const id = nanoid();
  localStorage.setItem(STORAGE_KEY, id);
  return id;
}
```

- Key is `playerId` per CONTEXT.md locked decision.
- Uses `nanoid()` (default 21 chars) for the token — not `nanoid(8)` (that's for room IDs).
- Called once at app startup; result is stable for the browser's lifetime.

**Create src/hooks/usePartySocket.ts:**

```typescript
import PartySocket from 'partysocket';
import { useEffect, useRef, useState } from 'react';
import type { ClientGameState, ServerEvent } from '../shared/types';

const PARTYKIT_HOST = import.meta.env.VITE_PARTYKIT_HOST
  ?? (import.meta.env.DEV ? 'localhost:1999' : 'localhost:1999');

export function usePartySocket(roomId: string, playerId: string) {
  const [gameState, setGameState] = useState<ClientGameState | null>(null);
  const [connected, setConnected] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const wsRef = useRef<PartySocket | null>(null);

  useEffect(() => {
    const ws = new PartySocket({
      host: PARTYKIT_HOST,
      room: roomId,
      id: playerId,
    });
    wsRef.current = ws;

    ws.addEventListener('open', () => {
      setConnected(true);
      setError(null);
    });

    ws.addEventListener('close', () => {
      setConnected(false);
    });

    ws.addEventListener('message', (e: MessageEvent) => {
      const event: ServerEvent = JSON.parse(e.data as string);
      if (event.type === 'STATE_UPDATE') {
        setGameState(event.state);
      } else if (event.type === 'ERROR') {
        setError(event.message);
      }
    });

    return () => {
      ws.close();
      wsRef.current = null;
    };
  }, [roomId, playerId]);

  return { gameState, connected, error };
}
```

Key details:
- `PARTYKIT_HOST` reads from `import.meta.env.VITE_PARTYKIT_HOST` first, falls back to `localhost:1999` for dev. Per Research Pitfall 2, the host MUST be env-gated.
- PartySocket `id` param is set to `playerId` — this becomes `connection.id` on the server, matching the Phase 1 player token approach.
- Returns `gameState` (latest ClientGameState from server), `connected` (boolean), `error` (latest error message or null).
- Uses `addEventListener` instead of `onmessage` assignment to avoid overwriting partysocket's internal handlers.
- `wsRef` stored for potential future use (sending messages) but not exposed yet.

```typescript
import { useEffect } from 'react';
import { nanoid } from 'nanoid';
import { getOrCreatePlayerId } from './hooks/usePlayerId';
import { usePartySocket } from './hooks/usePartySocket';
import LobbyPanel from './components/LobbyPanel';

export default function App() {
  const params = new URLSearchParams(window.location.search);
  const roomId = params.get('room');

  useEffect(() => {
    if (!roomId) {
      const id = nanoid(8);
      window.location.replace(`${window.location.pathname}?room=${id}`);
    }
  }, [roomId]);

  if (!roomId) return null;

  const playerId = getOrCreatePlayerId();
  const { gameState, connected, error } = usePartySocket(roomId, playerId);

  return (
    <LobbyPanel
      roomId={roomId}
      playerId={playerId}
      gameState={gameState}
      connected={connected}
      error={error}
    />
  );
}
```

Note: `getOrCreatePlayerId()` is called in render (not in a hook) because it's synchronous localStorage access — no effect needed. It returns the same value every time.

```yaml
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4

          node-version: lts/*
```
